#include<stdio.h>
#include<ctype.h>
#include<string.h>
int main(){
   char s1[100],s2[100];
   scanf("%s %s",s1,s2);
   int l1=strlen(s1),l2=strlen(s2);
   for(int i=0,j=0;i<l1,j<l2;i++,j++){
      s1[i]=toupper(s1[i]);
      s2[j]=toupper(s2[j]);
   }
   printf("%d\n",strcmp(s1,s2));
   return 0;
}
