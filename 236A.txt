    char str[100],copy[100]={0};
    scanf("%s",str);
    int len=strlen(str),l=0,i=0,j;
    while(str[i]!='\0'){
        for(j=0;j<len;j++){
            if(str[i]==copy[j]) break;
        }
        if(j==len) copy[l++]=str[i]; 
        i++;
    }
    int n=strlen(copy);
    //printf("%s %s",str,copy);
    if(n%2==0) printf("CHAT WITH HER!\n");
    else printf("IGNORE HIM!\n");
    return 0;
}
