#include<stdio.h>
int main(){
    int arr[5][5],a,b;
    for(int i=0;i<5;i++){
        for(int j=0;j<5;j++){
            scanf("%d",&arr[i][j]);
            if(arr[i][j]==1)
                a=i,b=j;
        }
    }
    int x=(a-2>0?a-2:2-a);
    int y=(b-2>0?b-2:2-b);
    printf("%d\n",x+y);
    return 0;
}
