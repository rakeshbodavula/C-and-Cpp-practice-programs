#include<stdio.h>
#include<string.h>
int main(){
    char arr[100];
    int str[50];
    scanf("%s",arr);
    int n=strlen(arr);
    for(int i=0;2*i<n;i++){
        str[i]=(int)arr[2*i];
    }
    int l=n/2;
    for(int i=0;i<=l;i++){
        for(int j=i+1;j<=l;j++){
            if(str[i]>str[j]){
                int a=str[i];
                str[i]=str[j];
                str[j]=a;
            }
        }
    }
    for(int i=0;i<=l;i++){
        if(i==l)
        printf("%c\n",str[i]);
        else
        printf("%c+",str[i]);
    }
    return 0;
}
